*** let
hyprland 클라이언트 창 바로가기를 관리하는 스크립트도 포함되어있다.
프로그램이 작동중인지 감지하고 작동중이라면 해당 창을 포커스하거나 해당하는 워크스페이스를 열어준다. 작동 중이 아니라면 해당 프로그램을 실행한다.
#+begin_src nix :tangle ./nixos/home/hypr.nix
  let
    inherit (pkgs.stdenv.hostPlatform) system;
    hyprpkgs = inputs.hyprland.packages."${system}";

    hypr-special-toggle = pkgs.writeShellApplication {
      name = "hypr-special-toggle";
      runtimeInputs = [ pkgs.jq pkgs.procps pkgs.fish ];
      text = ''
        <<fish_command>>
      '';
    };
  in
#+end_src

#+name: fish_command
#+begin_src fish :tangle no
  fish -c '
  set prog $argv[1]
  set ws $argv[2]

  # PID를 통해 프로그램의 class 를 추출
  set pids (pgrep -if $prog)
  set class = ""
  if test -n "$pids"
  set class (hypr)
#+end_src

*** hyprconf symlink
hyprland config를 nix로 굳이 다시 쓰기보다는 home-manager로 symlink만 해주고 설정 자체는 hyprlang으로 쓴다.

#+begin_src nix :tangle ./nixos/home/hypr.nix
  xdg.configFile.hypr = {
    source = config.lib.file.mkOutOfStoreSymlink "${config.home.homeDirectory}/repo/dotfiles/Configs/hyprland/.config/hypr";
  };
  };
  }
